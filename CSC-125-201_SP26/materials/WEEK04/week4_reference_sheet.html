Floor Division & Modulo in Practice
# Division always returns a float; // rounds down; % gives remainder
10 / 3     # 3.3333...
10 // 3    # 3
10 % 3     # 1     (relationship: 10 = 3*3 + 1)

# Time conversion
total_min = 135
hours = total_min // 60    # 2
mins = total_min % 60      # 15

# Even/odd check
number % 2 == 0    # True if even, False if odd

Boolean Expressions
age = 20
score = 85

can_vote = age >= 18           # True
is_passing = score >= 70       # True
is_perfect = score == 100      # False
is_teen = 13 <= age <= 19      # False (chained comparison)
